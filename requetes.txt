Toutes les reservations d'un utilisateur (Titre Ville Début fin annonceur )
USE lamp;
SELECT bookings.begin_at debut, bookings.end_at fin, announcements.title "titre annonce", announcements.city Ville, users.username propriaitraire
FROM bookings 
JOIN announcements ON bookings.announcement_id = announcements.idannouncement
JOIN users ON announcements.owner_id = users.iduser
WHERE users.iduser = ;
